136     private void Decode(byte buffer[], int shift, int[] out) {
137         /*
138          * len += shift; for (int i = 0; shift < len; i++, shift += 4) { out[i]
139          * = ((int) (buffer[shift] & 0xff)) | (((int) (buffer[shift + 1] &
140          * 0xff)) << 8) | (((int) (buffer[shift + 2] & 0xff)) << 16) | (((int)
141          * buffer[shift + 3]) << 24); }
142          */
143 
144         // unrolled loop (original loop shown above)
145         out[0] = buffer[shift] & 0xff | (buffer[shift + 1] & 0xff) << 8 |
146                 (buffer[shift + 2] & 0xff) << 16 | buffer[shift + 3] << 24;
147         out[1] = buffer[shift + 4] & 0xff | (buffer[shift + 5] & 0xff) << 8 |
148                 (buffer[shift + 6] & 0xff) << 16 | buffer[shift + 7] << 24;
149         out[2] = buffer[shift + 8] & 0xff | (buffer[shift + 9] & 0xff) << 8 |
150                 (buffer[shift + 10] & 0xff) << 16 | buffer[shift + 11] << 24;
151         out[3] = buffer[shift + 12] & 0xff | (buffer[shift + 13] & 0xff) << 8 |
152                 (buffer[shift + 14] & 0xff) << 16 | buffer[shift + 15] << 24;
153         out[4] = buffer[shift + 16] & 0xff | (buffer[shift + 17] & 0xff) << 8 |
154                 (buffer[shift + 18] & 0xff) << 16 | buffer[shift + 19] << 24;
155         out[5] = buffer[shift + 20] & 0xff | (buffer[shift + 21] & 0xff) << 8 |
156                 (buffer[shift + 22] & 0xff) << 16 | buffer[shift + 23] << 24;
157         out[6] = buffer[shift + 24] & 0xff | (buffer[shift + 25] & 0xff) << 8 |
158                 (buffer[shift + 26] & 0xff) << 16 | buffer[shift + 27] << 24;
159         out[7] = buffer[shift + 28] & 0xff | (buffer[shift + 29] & 0xff) << 8 |
160                 (buffer[shift + 30] & 0xff) << 16 | buffer[shift + 31] << 24;
161         out[8] = buffer[shift + 32] & 0xff | (buffer[shift + 33] & 0xff) << 8 |
162                 (buffer[shift + 34] & 0xff) << 16 | buffer[shift + 35] << 24;
163         out[9] = buffer[shift + 36] & 0xff | (buffer[shift + 37] & 0xff) << 8 |
164                 (buffer[shift + 38] & 0xff) << 16 | buffer[shift + 39] << 24;
165         out[10] = buffer[shift + 40] & 0xff | (buffer[shift + 41] & 0xff) << 8 |
166                 (buffer[shift + 42] & 0xff) << 16 | buffer[shift + 43] << 24;
167         out[11] = buffer[shift + 44] & 0xff | (buffer[shift + 45] & 0xff) << 8 |
168                 (buffer[shift + 46] & 0xff) << 16 | buffer[shift + 47] << 24;
169         out[12] = buffer[shift + 48] & 0xff | (buffer[shift + 49] & 0xff) << 8 |
170                 (buffer[shift + 50] & 0xff) << 16 | buffer[shift + 51] << 24;
171         out[13] = buffer[shift + 52] & 0xff | (buffer[shift + 53] & 0xff) << 8 |
172                 (buffer[shift + 54] & 0xff) << 16 | buffer[shift + 55] << 24;
173         out[14] = buffer[shift + 56] & 0xff | (buffer[shift + 57] & 0xff) << 8 |
174                 (buffer[shift + 58] & 0xff) << 16 | buffer[shift + 59] << 24;
175         out[15] = buffer[shift + 60] & 0xff | (buffer[shift + 61] & 0xff) << 8 |
176                 (buffer[shift + 62] & 0xff) << 16 | buffer[shift + 63] << 24;
177     }

179     private void Transform(MD5State stat, byte buffer[], int shift,
180             int[] decode_buf) {
181         int a = stat.state[0], b = stat.state[1], c = stat.state[2], d = stat.state[3], x[] = decode_buf;
182 
183         Decode(buffer, shift, decode_buf);
184 
185         /* Round 1 */
186         a += (b & c | ~b & d) + x[0] + 0xd76aa478; /* 1 */
187         a = (a << 7 | a >>> 25) + b;
188         d += (a & b | ~a & c) + x[1] + 0xe8c7b756; /* 2 */
189         d = (d << 12 | d >>> 20) + a;
190         c += (d & a | ~d & b) + x[2] + 0x242070db; /* 3 */
191         c = (c << 17 | c >>> 15) + d;
192         b += (c & d | ~c & a) + x[3] + 0xc1bdceee; /* 4 */
193         b = (b << 22 | b >>> 10) + c;
194 
195         a += (b & c | ~b & d) + x[4] + 0xf57c0faf; /* 5 */
196         a = (a << 7 | a >>> 25) + b;
197         d += (a & b | ~a & c) + x[5] + 0x4787c62a; /* 6 */
198         d = (d << 12 | d >>> 20) + a;
199         c += (d & a | ~d & b) + x[6] + 0xa8304613; /* 7 */
200         c = (c << 17 | c >>> 15) + d;
201         b += (c & d | ~c & a) + x[7] + 0xfd469501; /* 8 */
202         b = (b << 22 | b >>> 10) + c;
203 
204         a += (b & c | ~b & d) + x[8] + 0x698098d8; /* 9 */
205         a = (a << 7 | a >>> 25) + b;
206         d += (a & b | ~a & c) + x[9] + 0x8b44f7af; /* 10 */
207         d = (d << 12 | d >>> 20) + a;
208         c += (d & a | ~d & b) + x[10] + 0xffff5bb1; /* 11 */
209         c = (c << 17 | c >>> 15) + d;
210         b += (c & d | ~c & a) + x[11] + 0x895cd7be; /* 12 */
211         b = (b << 22 | b >>> 10) + c;
212 
213         a += (b & c | ~b & d) + x[12] + 0x6b901122; /* 13 */
214         a = (a << 7 | a >>> 25) + b;
215         d += (a & b | ~a & c) + x[13] + 0xfd987193; /* 14 */
216         d = (d << 12 | d >>> 20) + a;
217         c += (d & a | ~d & b) + x[14] + 0xa679438e; /* 15 */
218         c = (c << 17 | c >>> 15) + d;
219         b += (c & d | ~c & a) + x[15] + 0x49b40821; /* 16 */
220         b = (b << 22 | b >>> 10) + c;
221 
222         /* Round 2 */
223         a += (b & d | c & ~d) + x[1] + 0xf61e2562; /* 17 */
224         a = (a << 5 | a >>> 27) + b;
225         d += (a & c | b & ~c) + x[6] + 0xc040b340; /* 18 */
226         d = (d << 9 | d >>> 23) + a;
227         c += (d & b | a & ~b) + x[11] + 0x265e5a51; /* 19 */
228         c = (c << 14 | c >>> 18) + d;
229         b += (c & a | d & ~a) + x[0] + 0xe9b6c7aa; /* 20 */
230         b = (b << 20 | b >>> 12) + c;
231 
232         a += (b & d | c & ~d) + x[5] + 0xd62f105d; /* 21 */
233         a = (a << 5 | a >>> 27) + b;
234         d += (a & c | b & ~c) + x[10] + 0x02441453; /* 22 */
235         d = (d << 9 | d >>> 23) + a;
236         c += (d & b | a & ~b) + x[15] + 0xd8a1e681; /* 23 */
237         c = (c << 14 | c >>> 18) + d;
238         b += (c & a | d & ~a) + x[4] + 0xe7d3fbc8; /* 24 */
239         b = (b << 20 | b >>> 12) + c;
240 
241         a += (b & d | c & ~d) + x[9] + 0x21e1cde6; /* 25 */
242         a = (a << 5 | a >>> 27) + b;
243         d += (a & c | b & ~c) + x[14] + 0xc33707d6; /* 26 */
244         d = (d << 9 | d >>> 23) + a;
245         c += (d & b | a & ~b) + x[3] + 0xf4d50d87; /* 27 */
246         c = (c << 14 | c >>> 18) + d;
247         b += (c & a | d & ~a) + x[8] + 0x455a14ed; /* 28 */
248         b = (b << 20 | b >>> 12) + c;
249 
250         a += (b & d | c & ~d) + x[13] + 0xa9e3e905; /* 29 */
251         a = (a << 5 | a >>> 27) + b;
252         d += (a & c | b & ~c) + x[2] + 0xfcefa3f8; /* 30 */
253         d = (d << 9 | d >>> 23) + a;
254         c += (d & b | a & ~b) + x[7] + 0x676f02d9; /* 31 */
255         c = (c << 14 | c >>> 18) + d;
256         b += (c & a | d & ~a) + x[12] + 0x8d2a4c8a; /* 32 */
257         b = (b << 20 | b >>> 12) + c;
258 
259         /* Round 3 */
260         a += (b ^ c ^ d) + x[5] + 0xfffa3942; /* 33 */
261         a = (a << 4 | a >>> 28) + b;
262         d += (a ^ b ^ c) + x[8] + 0x8771f681; /* 34 */
263         d = (d << 11 | d >>> 21) + a;
264         c += (d ^ a ^ b) + x[11] + 0x6d9d6122; /* 35 */
265         c = (c << 16 | c >>> 16) + d;
266         b += (c ^ d ^ a) + x[14] + 0xfde5380c; /* 36 */
267         b = (b << 23 | b >>> 9) + c;
268 
269         a += (b ^ c ^ d) + x[1] + 0xa4beea44; /* 37 */
270         a = (a << 4 | a >>> 28) + b;
271         d += (a ^ b ^ c) + x[4] + 0x4bdecfa9; /* 38 */
272         d = (d << 11 | d >>> 21) + a;
273         c += (d ^ a ^ b) + x[7] + 0xf6bb4b60; /* 39 */
274         c = (c << 16 | c >>> 16) + d;
275         b += (c ^ d ^ a) + x[10] + 0xbebfbc70; /* 40 */
276         b = (b << 23 | b >>> 9) + c;
277 
278         a += (b ^ c ^ d) + x[13] + 0x289b7ec6; /* 41 */
279         a = (a << 4 | a >>> 28) + b;
280         d += (a ^ b ^ c) + x[0] + 0xeaa127fa; /* 42 */
281         d = (d << 11 | d >>> 21) + a;
282         c += (d ^ a ^ b) + x[3] + 0xd4ef3085; /* 43 */
283         c = (c << 16 | c >>> 16) + d;
284         b += (c ^ d ^ a) + x[6] + 0x04881d05; /* 44 */
285         b = (b << 23 | b >>> 9) + c;
286 
287         a += (b ^ c ^ d) + x[9] + 0xd9d4d039; /* 33 */
288         a = (a << 4 | a >>> 28) + b;
289         d += (a ^ b ^ c) + x[12] + 0xe6db99e5; /* 34 */
290         d = (d << 11 | d >>> 21) + a;
291         c += (d ^ a ^ b) + x[15] + 0x1fa27cf8; /* 35 */
292         c = (c << 16 | c >>> 16) + d;
293         b += (c ^ d ^ a) + x[2] + 0xc4ac5665; /* 36 */
294         b = (b << 23 | b >>> 9) + c;
295 
296         /* Round 4 */
297         a += (c ^ (b | ~d)) + x[0] + 0xf4292244; /* 49 */
298         a = (a << 6 | a >>> 26) + b;
299         d += (b ^ (a | ~c)) + x[7] + 0x432aff97; /* 50 */
300         d = (d << 10 | d >>> 22) + a;
301         c += (a ^ (d | ~b)) + x[14] + 0xab9423a7; /* 51 */
302         c = (c << 15 | c >>> 17) + d;
303         b += (d ^ (c | ~a)) + x[5] + 0xfc93a039; /* 52 */
304         b = (b << 21 | b >>> 11) + c;
305 
306         a += (c ^ (b | ~d)) + x[12] + 0x655b59c3; /* 53 */
307         a = (a << 6 | a >>> 26) + b;
308         d += (b ^ (a | ~c)) + x[3] + 0x8f0ccc92; /* 54 */
309         d = (d << 10 | d >>> 22) + a;
310         c += (a ^ (d | ~b)) + x[10] + 0xffeff47d; /* 55 */
311         c = (c << 15 | c >>> 17) + d;
312         b += (d ^ (c | ~a)) + x[1] + 0x85845dd1; /* 56 */
313         b = (b << 21 | b >>> 11) + c;
314 
315         a += (c ^ (b | ~d)) + x[8] + 0x6fa87e4f; /* 57 */
316         a = (a << 6 | a >>> 26) + b;
317         d += (b ^ (a | ~c)) + x[15] + 0xfe2ce6e0; /* 58 */
318         d = (d << 10 | d >>> 22) + a;
319         c += (a ^ (d | ~b)) + x[6] + 0xa3014314; /* 59 */
320         c = (c << 15 | c >>> 17) + d;
321         b += (d ^ (c | ~a)) + x[13] + 0x4e0811a1; /* 60 */
322         b = (b << 21 | b >>> 11) + c;
323 
324         a += (c ^ (b | ~d)) + x[4] + 0xf7537e82; /* 61 */
325         a = (a << 6 | a >>> 26) + b;
326         d += (b ^ (a | ~c)) + x[11] + 0xbd3af235; /* 62 */
327         d = (d << 10 | d >>> 22) + a;
328         c += (a ^ (d | ~b)) + x[2] + 0x2ad7d2bb; /* 63 */
329         c = (c << 15 | c >>> 17) + d;
330         b += (d ^ (c | ~a)) + x[9] + 0xeb86d391; /* 64 */
331         b = (b << 21 | b >>> 11) + c;
332 
333         stat.state[0] += a;
334         stat.state[1] += b;
335         stat.state[2] += c;
336         stat.state[3] += d;
337     }

339     /**
340      * Updates hash with the bytebuffer given (using at maximum length bytes
341      * from that buffer)
342      *
343      * @param stat
344      *            Which state is updated
345      * @param buffer
346      *            Array of bytes to be hashed
347      * @param offset
348      *            Offset to buffer array
349      * @param length
350      *            Use at maximum `length' bytes (absolute maximum is
351      *            buffer.length)
352      */
353     public void Update(MD5State stat, byte buffer[], int offset, int length) {
354         int index, partlen, i, start;
355         finals = null;
356         int newlength = length;
357         /* Length can be told to be shorter, but not inter */
358         if (newlength + offset > buffer.length) {
359             newlength = buffer.length - offset;
360         }
361 
362         /* compute number of bytes mod 64 */
363 
364         index = (int) (stat.count & 0x3f);
365         stat.count += newlength;
366 
367         partlen = 64 - index;
368 
369         if (newlength >= partlen) {
370 			// update state (using only Java) to reflect input
371 
372 			int[] decode_buf = new int[16];
373 			if (partlen == 64) {
374 				partlen = 0;
375 			} else {
376 				for (i = 0; i < partlen; i++) {
377 					stat.buffer[i + index] = buffer[i + offset];
378 				}
379 				Transform(stat, stat.buffer, 0, decode_buf);
380 			}
381 			for (i = partlen; i + 63 < newlength; i += 64) {
382 				Transform(stat, buffer, i + offset, decode_buf);
383 			}
384             index = 0;
385         } else {
386             i = 0;
387         }
388 
389         /* buffer remaining input */
390         if (i < newlength) {
391             start = i;
392             for (; i < newlength; i ++) {
393                 stat.buffer[index + i - start] = buffer[i + offset];
394             }
395         }
396     }

448     /**
449      * Update buffer with given string. Note that because the version of the
450      * s.getBytes() method without parameters is used to convert the string to a
451      * byte array, the results of this method may be different on different
452      * platforms. The s.getBytes() method converts the string into a byte array
453      * using the current platform's default character set and may therefore have
454      * different results on platforms with different default character sets. If
455      * a version that works consistently across platforms with different default
456      * character sets is desired, use the overloaded version of the Update()
457      * method which takes a string and a character encoding.
458      *
459      * @param s
460      *            String to be update to hash (is used as s.getBytes())
461      */
462     public void Update(String s) {
463         byte chars[] = s.getBytes();
464         Update(chars, chars.length);
465     }

467     /**
468      * Update buffer with given string using the given encoding. If the given
469      * encoding is null, the encoding "ISO8859_1" is used.
470      *
471      * @param s
472      *            String to be update to hash (is used as
473      *            s.getBytes(charset_name))
474      * @param charset_name
475      *            The character set to use to convert s to a byte array, or null
476      *            if the "ISO8859_1" character set is desired.
477      * @exception java.io.UnsupportedEncodingException
478      *                If the named charset is not supported.
479      */
480     public void Update(String s, String charset_name)
481             throws java.io.UnsupportedEncodingException {
482         String newcharset = charset_name;
483         if (newcharset == null) {
484             newcharset = "ISO8859_1";
485         }
486         byte chars[] = s.getBytes(newcharset);
487         Update(chars, chars.length);
488     }

500     /**
501      * Updates hash with given {@link ChannelBuffer} (from Netty)
502      *
503      * @param buffer
504      *            ChannelBuffer to use for updating the hash
505      *            and this buffer will not be changed
506      */
507     public void Update(ChannelBuffer buffer) {
508         byte[] bytes;
509         int start = 0;
510         int len = buffer.readableBytes();
511         if (buffer.hasArray()) {
512             start = buffer.arrayOffset();
513             bytes = buffer.array();
514             if (bytes.length > start+len) {
515                 byte[] temp = new byte[len];
